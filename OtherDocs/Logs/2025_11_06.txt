        import requests
        import pandas as pd

        def get_member_info(name):
            url = f"https://api.oireachtas.ie/v1/members?search={name}"
            response = requests.get(url)
            data = response.json()

            if not data['results']:
                return None

            member = data['results'][0]['member']
            info = {
                'name': member['fullName'],
                'constituency': member.get('membership', [{}])[0].get('membership', {}).get('showAs', 'Unknown'),
                'gender': member.get('gender', {}).get('showAs', 'Unknown')
            }
            return info
